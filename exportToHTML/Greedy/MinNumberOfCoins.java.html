13       static int[] denominations = {1, 2, 5, 10, 20, 50, 500, 1000};
14       static int n = denominations.length;
15       static List<Integer> ans = new LinkedList<>();
16   
17       static void findMinCoin(int V) {
18   
19           for(int i=n-1; i>=0; i--) {
20   
21               while(V >= denominations[i]) {
22                   V -= denominations[i];
23                   ans.add(denominations[i]);
24               }
25           }
26   
27           System.out.println(ans);
28       }
