1    package Greedy;
2    import java.util.*;
3    
4    // this approach does not work with -ve values
5    class Activity {
6        int start, finish;
7    
8        public Activity(int start, int finish) {
9            this.start = start;
10           this.finish = finish;
11       }
12   }
13   
14   public class ActivitySelection {
15       static void printMaxActivities(Activity arr[], int n) {
16   
17           Arrays.sort(arr, new Comparator<Activity>() {
18               @Override
19               public int compare(Activity s1, Activity s2) {
20                   return s1.finish - s2.finish;
21               }
22           });
23           
24           System.out.println("Following activities are selected : ");
25   
26           int i = 0;
27           System.out.print("(" +
28                   arr[i].start + ", " +
29                   arr[i].finish +
30                   "), ");
31   
32           for (int j = 1; j < n; j++) {
33               if (arr[j].start >= arr[i].finish) {
34                   System.out.print("(" + arr[j].start + ", "
35                           + arr[j].finish + "), ");
36                   i = j;
37               }
38           }
39       }
40   
41       public static void main(String[] args) {
42   
43           int n = 6;
44           Activity arr[] = new Activity[n];
45           arr[0] = new Activity(5, 9);
46           arr[1] = new Activity(1, 2);
47           arr[2] = new Activity(3, 4);
48           arr[3] = new Activity(0, 6);
49           arr[4] = new Activity(5, 7);
50           arr[5] = new Activity(8, 9);
51   
52           printMaxActivities(arr, n);
53       }
54   }
