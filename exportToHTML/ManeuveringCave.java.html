3    // to count all the possible paths from top-left to bottom-right of a NxM matrix
4    // with rules that either move only right or down
5    
6    // f(m, n) = it can be divided into 2 sub-problems = f(m-1, n) + f(m, n-1) in solving the cave
7    
8    public class ManeuveringCave {
9        public static void main(String[] args) {
10           System.out.println(numberOfPaths(3, 3));
11       }
12   
13       private static int numberOfPaths(int m, int n) {
14   
15           // if only one row then have only 1 direction
16           // if only 1 column then have only 1 direction
17           // base condition
18           if (m == 1 || n == 1)
19               return 1;
20   
21           return numberOfPaths(m - 1, n) + numberOfPaths(m, n - 1);
22       }
23   }
