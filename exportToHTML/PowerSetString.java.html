1    // power set includes all possible pairs of chars in given string
2    // ex: abc = {"", a, b, c, ab, ac, bc, abc}
3    
4    import java.util.ArrayList;
5    import java.util.List;
6    
7    public class PowerSetString {
8        public static void main(String[] args) {
9    
10           String s = "abc";
11           List<String> result = new ArrayList<>();
12           powerSet(s, 0, "", result);
13           System.out.println(result);
14       }
15   
16       // using recursion: time complexity = O(2^n)
17       private static void powerSet(String s, int i, String current, List<String> result) {
18   
19           // base condition
20           if (i == s.length()) {
21               result.add(current);
22               return;
23           }
24   
25           powerSet(s, i + 1, current + s.charAt(i), result);
26           powerSet(s, i + 1, current, result);
27       }
28   }
