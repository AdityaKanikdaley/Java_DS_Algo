1    // to see if the number is Strobogrammatic or not
2    // if a number contain numbers other than 0, 1, 8, 6, 9 then its non Strobogrammatic
3    // but if number contains: 6 or 9 : convert 6->9 & 9->6
4    // test cases: 8698, 9696, 1691, 6795, 1221
5    
6    public class Strobogrammatic {
7    
8        public static void main(String[] args) {
9            int num = 9898;
10           System.out.println(solve(num));
11       }
12   
13       private static boolean solve(int num) {
14   
15           int temp = num;
16           int revNum = 0;
17           while (temp > 0) {
18               int lastDigit = temp % 10;
19   
20               if (lastDigit == 0 || lastDigit == 1 || lastDigit == 8 || lastDigit == 6 || lastDigit == 9) {
21                   if (lastDigit == 6) lastDigit = 9;
22                   else if (lastDigit == 9) lastDigit = 6;
23               } else {
24                   return false;
25               }
26   
27               revNum = (revNum * 10) + lastDigit;
28               temp /= 10;
29           }
30   
31           System.out.println("Num: " + num);
32           System.out.println("Rev: " + revNum);
33           return revNum == num;
34       }
35   }
