2    formula: (a+b) % n = (a%n + b%n) % n 
3             (a*b) % n = (a%n * b%n) % n 
4    */
5    public class ModuloArithmetics {
6        public static void main(String[] args) {

8        }

10       private static long fastPower(long a, long b, int n) {
11   
12           long result = 1;
13           while (b > 0) {
14               if ((b & 1) != 0)  // check if b is odd
15                   result = (result * a % n) % n; // (result % n * a % n) % n = (result * a % n) % n as result < n, so its modulo will be result itself
16   
17               a = (a % n * a % n) % n;
18               b = b >> 1; // divide by 2
19           }
20           return result;
21       }
22   }
