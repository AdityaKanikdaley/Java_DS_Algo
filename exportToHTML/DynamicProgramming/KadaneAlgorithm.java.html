3    // Kadane's algorithm : find the largest sum in contiguous subArray
4    
5    class KadaneAlgorithm {
6    
7        // this approach handles when all elements in array are negative
8        static int maxSubArraySum(int[] a, int size)
9        {
10           int max_so_far = a[0];
11           int current_max = a[0];
12   
13           for (int i = 1; i < size; i++)
14           {
15               current_max = Math.max(a[i], current_max + a[i]);
16               max_so_far = Math.max(max_so_far, current_max);
17           }
18           return max_so_far;
19       }
