8    public class LongestIncreasingSubsequence {
9        public static void main(String[] args) {
10           
11           int[] nums = {10,9,2,5,3,7,101,18};
12   
13           int n = nums.length;
14           
15           // creating dp array
16           int t[] = new int[n];
17           Arrays.fill(t, 1);  // every number in itself is a LIS of length 1
18           
19           // O(n_square) solution
20           // t[i] -> is the length of LIS till i
21           for(int i = 1; i < n; i++) {
22               // simply iterating over array till 'i', and updating 't' array
23               for(int j = 0; j < i ; j++) {
24                   if(nums[j] < nums[i]) {
25                       t[i] = Math.max(t[i], t[j] + 1);
26                   }
27               }
28           }
29           
30           // finding maximum in the dp array 't'
31           int LIS_length = 0;
32           for(int v : t) {
33              if(v > LIS_length) {
34                  LIS_length = v;
35              }
36           }
37   
38           System.out.println(LIS_length);
39       }
40   }
