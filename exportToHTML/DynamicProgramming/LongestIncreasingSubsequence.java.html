49   public class LongestIncreasingSubsequence {
50       public static void main(String[] args) {
51           
52           int[] nums = {10,9,2,5,3,7,101,18};
53   
54           int overallMax = 0;
55           int dp[] = new int[nums.length];
56   
57           for(int i=0; i<nums.length; i++) {
58               int max = 0;
59               
60               for(int j=0; j<i; j++) {
61                   if(nums[i] > nums[j])
62                       if(dp[j] > max)
63                           max = dp[j];
64               }
65               
66               dp[i] = max + 1;
67               
68               if(dp[i] > overallMax)
69                   overallMax = dp[i];
70           }
71   
72           System.out.println(overallMax);
73       }
74   }
