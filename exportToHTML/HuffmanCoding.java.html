1    import java.util.Comparator;
2    import java.util.PriorityQueue;
3    
4    // node class is the basic structure
5    // of each node present in the Huffman - tree.
6    class HuffmanNode {
7        int data;
8        char c;
9    
10       HuffmanNode left;
11       HuffmanNode right;
12   }
13   
14   // comparator class helps to compare the node
15   // on the basis of one of its attribute.
16   // Here we will be compared
17   // on the basis of data values of the nodes.
18   class MyComparator implements Comparator<HuffmanNode> {
19       public int compare(HuffmanNode x, HuffmanNode y) {
20           return x.data - y.data;
21       }
22   }
23   
24   public class HuffmanCoding {
25       
26       // recursive function to print the
27       // huffman-code through the tree traversal.
28       // Here s is the huffman - code generated.
29       public static void printCode(HuffmanNode root, String s) {
30   
31           // base case; if the left and right are null
32           // then its a leaf node and we print
33           // the code s generated by traversing the tree.
34           if(root.left == null && root.right == null && Character.isLetter(root.c)) {
35               System.out.println(root.c + ":" + s);
36               return;
37           }
38   
39           // if we go to left then add "0" to the code.
40           // if we go to the right add"1" to the code.
41   
42           // recursive calls for left and
43           // right sub-tree of the generated tree.
44           printCode(root.left, s + "0");
45           printCode(root.right, s + "1");
46       }
47   
48   
49           // main function
50           public static void main(String[] args) {
51               
52               int n = 6;
53               char[] charArray = {'a', 'b', 'c', 'd', 'e', 'f'};
54               int[] charFreq = {5, 9, 12, 13, 16, 45};
55   
56               // creating a priority queue q.
57               // makes a min-priority queue(min-heap).
58               PriorityQueue<HuffmanNode> q = new PriorityQueue<>(n, new MyComparator());
59   
60               for(int i=0; i<n; i++) {
61                   
62                   // creating a Huffman node object
63                   // and add it to the priority queue.
64                   HuffmanNode hn = new HuffmanNode();
65                   hn.c = charArray[i];
66                   hn.data = charFreq[i];
67   
68                   q.add(hn);
69               }
70   
71               // create a root node
72               HuffmanNode root = null;
73               
74               // Here we will extract the two minimum value
75               // from the heap each time until
76               // its size reduces to 1, extract until
77               // all the nodes are extracted.
78   
79               while(q.size() > 1) {
80   
81                   HuffmanNode x = q.poll(); // first min extract.
82                   HuffmanNode y = q.poll(); // second min extract.
83   
84                   // new node f which is equal
85                   // to the sum of the frequency of the two nodes
86                   // assigning values to the f node.  
87                   HuffmanNode f = new HuffmanNode();
88   
89                   f.data = x.data + y.data;
90                   f.c = '-';
91   
92                   // first extracted node as left child.
93                   // second extracted node as the right child.
94                   f.left = x;
95                   f.right = y;
96   
97                   // marking the f node as the root node.
98                   root = f;
99   
100                  // add this node to the priority-queue.
101                  q.add(f);
102              }
103  
104              printCode(root, "");
105          }
106  }
