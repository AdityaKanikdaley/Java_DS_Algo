3    /* 
4    formula of GCD = GCD(a, b) = GCD(a-b, b) , a>b 
5               GCD(a, b) = GCD(b, a%b), till a%b != 0 
6    */
7    
8    public class EuclidGCD {
9        public static void main(String[] args) {
10           System.out.println(e_gcd(24, 60));
11       }
12   
13       private static int e_gcd(int a, int b) {
14   
15           // base condition
16           if (b == 0) return a;
17   
18           //  here:   a   b
19           return e_gcd(b, a % b);
20       }
21   }
