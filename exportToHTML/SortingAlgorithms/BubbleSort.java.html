15   public class BubbleSort {
16       public static void main(String[] args) {
17           int[] arr = {64, 34, 25, 12, 22, 11, 90};
18           bubbleSort(arr);
19           System.out.println(Arrays.toString(arr));
20       }
21   
22       private static void bubbleSort(int[] arr) {
23           int n = arr.length;
24           for (int i = 0; i < n - 1; i++)
25               for (int j = 0; j < n - i - 1; j++)
26                   if (arr[j + 1] < arr[j])
27                       swap(arr, j + 1, j);
28       }
29   
30       private static void swap(int[] arr, int a, int b) {
31           int temp = arr[a];
32           arr[a] = arr[b];
33           arr[b] = temp;
34       }
35   }
