6    In insertion sort, we do fewer operations than bubble sort, 
7    that's why we also call insertion sort is an optimization over bubble sort. 
8     
9    tc: O(n) in best case 
10       O(n^2) in average case 
11       O(n^2) in worst case 
12    
13   sc: O(1)

16   import java.util.Arrays;
17   
18   public class InsertionSort {
19       public static void main(String[] args) {
20           int[] arr = {4, 8, 1, 3, 2, 7};
21           insertionSort(arr);
22           System.out.println(Arrays.toString(arr));
23       }
24   
25       private static void insertionSort(int[] arr) {
26           int n = arr.length;
27           for (int i = 1; i < n; i++) {
28               int temp = arr[i];
29               int j = i - 1;
30               while (j >= 0 && arr[j] > temp) {
31                   arr[j + 1] = arr[j]; // right shifting of elements
32                   j--;
33               }
34   
35               arr[j + 1] = temp; // assigning temp to the space caused due to right shift
36           }
37       }
38   }
