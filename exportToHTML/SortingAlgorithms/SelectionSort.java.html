15   public class SelectionSort {
16       public static void main(String[] args) {
17           int[] arr = {4, 8, 1, 3, 2, 7};
18           selectionSort(arr);
19           System.out.println(Arrays.toString(arr));
20       }
21   
22       private static void selectionSort(int[] arr) {
23           // "min", "i", "j" -> these are indices not values
24           int n = arr.length;
25           for (int i = 0; i < n - 1; i++) {
26               int min = i;
27               for (int j = i + 1; j < n; j++) {
28                   if (arr[j] < arr[min]) {
29                       min = j;
30                   }
31               }
32   
33               if (min != i) // to prevent swapping from itself (extra swap)
34                   swap(arr, min, i);
35           }
36       }
37   
38       private static void swap(int[] arr, int a, int b) {
39           int temp = arr[a];
40           arr[a] = arr[b];
41           arr[b] = temp;
42       }
43   }
