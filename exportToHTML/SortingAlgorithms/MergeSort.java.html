15   public class MergeSort {
16       public static void main(String[] args) {
17           int[] arr = {9, 4, 7, 6, 3, 1, 5};
18           mergeSort(arr, 0, arr.length - 1);
19           System.out.println(Arrays.toString(arr));
20       }
21   
22       private static void mergeSort(int[] arr, int l, int r) {
23           if (l < r) { // continue only if left < right -> array have at least 2 elements
24               int mid = (l + r) / 2;
25               mergeSort(arr, l, mid); // call mergeSort for left half
26               mergeSort(arr, mid + 1, r); // call mergeSort for right half
27               merge(arr, l, mid, r); // after recursion, merge left and right halves
28           }
29       }
30   
31       private static void merge(int[] arr, int l, int mid, int r) {
32           int[] b = new int[arr.length]; // new temp array
33   
34           int i = l;
35           int j = mid + 1;
36           int k = l;
37   
38           while (i <= mid && j <= r) { // i will go till mid only & j will go only till right
39               if (arr[i] < arr[j]) { // if arr[i] < arr[j] -> take that ith element and i++
40                   b[k] = arr[i];
41                   i++;
42               } else { // otherwise take jth element and j++
43                   b[k] = arr[j];
44                   j++;
45               }
46               k++; // k++ to go next index in "b[]"
47           }
48   
49           if (i > mid) { // if left half exhausted and right half is still have some elements -> copy those right half elements as it is in "b[]"
50               while (j <= r) {
51                   b[k] = arr[j];
52                   k++;
53                   j++;
54               }
55           } else { // if right half exhausted and left half still have some elements -> copy those left elements as it is in "b[]"
56               while (i <= mid) {
57                   b[k] = arr[i];
58                   k++;
59                   i++;
60               }
61           }
62   
63           for (k = l; k <= r; k++) { // copy all elements from temp array "b[]" to "arr[]"
64               arr[k] = b[k];
65           }
66       }
67   }
