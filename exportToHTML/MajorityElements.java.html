13       public static int majorityElement(int[] nums) {
14           int count = 0;
15           Integer candidate = null;
16   
17           for (int num : nums) {
18               if (count == 0) {
19                   candidate = num;
20               }
21               count += (num == candidate) ? 1 : -1;
22           }
23   
24           return candidate;
25       }
