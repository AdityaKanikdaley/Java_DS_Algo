2    import java.util.LinkedList;
3    import java.util.List;

6    public class PermutationString {
7    
8        static List<String> permutations = new LinkedList<>();
9    
10       public static void main(String[] args) {
11           String s = "abc";
12           permute(s, 0, s.length());
13           System.out.println("No. of permutations: " + permutations.size());
14           System.out.println("Permutations: " + permutations);
15       }
16   
17       private static void permute(String s, int l, int r) {
18   
19           // base condition
20           if (l == r) {
21               permutations.add(s);
22               return;
23           }
24   
25           for (int i = l; i < r; i++) {
26               s = swap(s, l, i);
27               permute(s, l + 1, r);
28               s = swap(s, l, i); // backtracking
29           }
30       }
31   
32       private static String swap(String s, int l, int i) {
33   
34           char[] arr = s.toCharArray();
35           char temp = arr[l];
36           arr[l] = arr[i];
37           arr[i] = temp;
38   
39           return new String(arr);
40       }
41   }
