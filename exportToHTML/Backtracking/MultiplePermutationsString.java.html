22       private static List<String> cal_permute(String str) {
23   
24           List<String> permutation = new ArrayList<>();
25           solve(permutation, str, 0, str.length());
26   
27           return permutation;
28       }

30       private static void solve(List<String> permutation, String str, int l, int r) {
31   
32           if(l == r) {
33               permutation.add(str);
34               return;
35           }
36   
37           for(int i=l; i<r; i++) {
38               str = swap(str, l, i);
39               solve(permutation, str, l+1, r);
40               str = swap(str, l, i);
41           }
42       }
43   
44       private static String swap(String str, int a, int b) {
45   
46           char[] ch = str.toCharArray();
47           char temp = ch[a];
48           ch[a] = ch[b];
49           ch[b] = temp;
50   
51           return new String(ch);
52       }
