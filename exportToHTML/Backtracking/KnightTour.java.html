1    package Backtracking;
2    public class KnightTour {
3    
4        static int N = 8;
5        static boolean isSafe(int x, int y, int[][] sol) {
6            return (x >= 0 && x < N && y >= 0 && y < N && sol[x][y] == -1);
7        }
8    
9        static void printSolution(int[][] sol) {
10           for (int x = 0; x < N; x++) {
11               for (int y = 0; y < N; y++)
12                   System.out.print(sol[x][y] + " ");
13               System.out.println();
14           }
15       }
16   
17       static void solveKT() {
18           int[][] sol = new int[8][8];
19   
20           /* Initialization of solution matrix */
21           for (int x = 0; x < N; x++)
22               for (int y = 0; y < N; y++)
23                   sol[x][y] = -1;
24   
25           // going clockwise
26           int[] xMove = {-2, -1, +1, +2, +2, +1, -1, -2};
27           int[] yMove = {+1, +2, +2, +1, -1, -2, -2, -1};
28   
29           // Since the Knight is initially at the first block
30           sol[0][0] = 0;
31   
32           // Start from 0,0 and explore all tours using solveKTUtil()
33           if (!solveKTUtil(0, 0, 1, sol, xMove, yMove))
34               System.out.println("Solution does not exist");
35            else
36               printSolution(sol);
37       }
38   
39       // A recursive utility function to solve Knight Tour problem
40       static boolean solveKTUtil(int x, int y, int move, int[][] sol, int[] xMove, int[] yMove) {
41           int next_x, next_y;
42   
43           if (move == N * N) return true;
44   
45           // Try all next moves from the current coordinate x, y
46           for (int k = 0; k < 8; k++) {
47               next_x = x + xMove[k];
48               next_y = y + yMove[k];
49   
50               if (isSafe(next_x, next_y, sol)) {
51                   sol[next_x][next_y] = move;
52   
53                   if (solveKTUtil(next_x, next_y, move + 1, sol, xMove, yMove))
54                       return true;
55                   else
56                       sol[next_x][next_y] = -1; // backtracking
57               }
58           }
59           return false;
60       }
61   
62       public static void main(String[] args) {
63           solveKT();
64       }
65   }
