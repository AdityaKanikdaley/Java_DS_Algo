1    import java.util.ArrayList;

4    public class PowerSetArray {
5        public static void main(String[] args) {
6            int[] arr = {1, 2, 3};
7            List<List<Integer>> result = new ArrayList<>();
8            getSubsets(arr, result, new ArrayList<>(), 0);
9            for (List<Integer> l : result)
10               System.out.println(l);
11       }
12   
13       private static void getSubsets(int[] arr, List<List<Integer>> result, List<Integer> curList, int start) {
14           result.add(new ArrayList<>(curList));
15           for (int i = start; i < arr.length; i++) {
16               curList.add(arr[i]);
17               getSubsets(arr, result, curList, i + 1);
18               curList.remove(curList.size() - 1);
19           }
20       }
