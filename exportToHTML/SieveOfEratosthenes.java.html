9        private static void s_o_e(int n) {
10   
11           // Create a boolean array "prime[0..n]" and initialize
12           // all entries it as true. A value in prime[i] will
13           // finally be false if (i) is Not a prime, else true.
14           boolean[] primes = new boolean[n + 1];
15   
16           // fill all values to true
17           Arrays.fill(primes, true);
18           for (int p = 2; p * p <= n; p++) {
19   
20               // If prime[p] is not changed, then it is a prime
21               if (primes[p] == true) {
22   
23                   // Update all multiples of p
24                   for (int i = p * p; i <= n; i += p)
25                       primes[i] = false;
26               }
27           }
28   
29           System.out.println("Primes till " + n);
30           // Print all prime numbers
31           for (int i = 2; i <= n; i++)
32               if (primes[i] == true)
33                   System.out.print(i + " ");
34       }
35   }
