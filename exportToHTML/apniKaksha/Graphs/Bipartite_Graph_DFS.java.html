45       static boolean dfsCheck(ArrayList < ArrayList < Integer >> graph, int node, int[] color) {
46           for (Integer it: graph.get(node)) {
47               if (color[it] == -1) {
48                   //Color of variable neighbor is the inverted color of variable node
49                   color[it] = 1 - color[node];
50   
51                   if (!dfsCheck(graph, it, color))
52                       return false;
53               } else if (color[it] == color[node]) {
54                   return false;
55               }
56           }
57           return true;
58       }
59   
60       static boolean checkBipartite(ArrayList < ArrayList < Integer >> graph, int n) {
61           int[] color = new int[n];
62   
63           for (int i = 0; i < n; i++) {
64               color[i] = -1;
65           }
66   
67           for (int i = 0; i < n; i++) {
68               if (color[i] == -1) {
69                   if (!dfsCheck(graph, i, color)) {
70                       return false;
71                   }
72               }
73           }
74           return true;
75       }
