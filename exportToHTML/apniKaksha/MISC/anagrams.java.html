3    // anagram = 2 strings in which number of characters are repeats same number of time
4    // example = aab & baa are anagram, but bad & gfr are not anagram
5    
6    public class anagrams {
7        public static void main(String[] args) {
8    
9            String a = "aab";
10           String b = "aba";
11   
12           // method 1 = O(n^2)
13           boolean isAnagram = false;
14           boolean[] isVisited = new boolean[b.length()];
15   
16           if(a.length() == b.length()){
17               for (int i=0; i<a.length(); i++){
18                   char check = a.charAt(i);
19                   isAnagram = false;
20                   for (int j=0; j<b.length(); j++){
21                       if (b.charAt(j) == check && !isVisited[j]){
22                           isVisited[j] = true;
23                           isAnagram = true;
24                           break;
25                       }
26                   }
27                   if (!isAnagram){
28                       break;
29                   }
30               }
31           }
32   
33           // method 2 = O(n)
34   
35           boolean isAnagram = true;
36           int[] aLength = new int[256];
37           int[] bLength = new int[256];
38   
39           for (char check : a.toCharArray()){
40               int index = (int) check;
41               aLength[index]++;
42           }
43   
44           for (char check : b.toCharArray()){
45               int index = (int) check;
46               bLength[index]++;
47           }
48   
49           for (int i=0; i<256; i++){
50               if(aLength[i] != bLength[i]){
51                   isAnagram = false;
52                   break;
53               }
54           }
55   
56           // method 3 = O(n), optimized of  method 2
57   
58           boolean isAnagram = true;
59           int[] aLength = new int[256];
60   
61           for (char check : a.toCharArray()) {
62               int index = check;
63               aLength[index]++;
64           }
65   
66           for (char check : b.toCharArray()) {
67               int index = check;
68               aLength[index]--;
69           }
70   
71           for (int i = 0; i < 256; i++) {
72               if (aLength[i] != 0) {
73                   isAnagram = false;
74                   break;
75               }
76           }
77           if (isAnagram)
78               System.out.println("Anagram");
79           else
80               System.out.println("Not anagram");
81       }
82   }
